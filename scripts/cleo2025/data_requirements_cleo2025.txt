# data_requirements_cleo2025.txt — CLEO 2025
#
# Structured manifest of the data files that the fetch (downloader) and process
# scripts in this directory rely on. make_app.py reads this file to decide
# whether a download can be skipped: when FORCE_DOWNLOAD is off, every file
# marked "required = yes" below must already exist in data/ for the download to
# be skipped; otherwise make_app runs the downloader.
#
# Format (simple, line-oriented; '#' starts a comment, blank lines ignored):
#   - Each file is a block introduced by a header line "[file: <pattern>]".
#   - <pattern> is a filename or a glob (e.g. *_Program_Abstracts.pdf) resolved
#     relative to the data/ subdirectory of THIS conference folder.
#   - Inside a block, "key: value" lines give metadata. Recognized keys:
#       required     yes | no   (default yes) — whether the file must be present
#       produced_by  which script normally creates it (fetch / process / manual)
#       description  one-line human description
#       manual       how to obtain the file by hand if the downloader can't
#   - Continuation: a "manual" value may wrap onto following indented lines.
#
# manual_steps_written: 2026-05-24
# These manual-download instructions were accurate as of the date above; the
# conference website / planner layout can change, so re-verify if much later.

generated_for: cleo2025
manual_steps_written: 2026-05-24

[file: CLEO2025_Program_Abstracts.pdf]
required: yes
produced_by: fetch
description: Official "Program + Abstracts (PDF)" — full program with every
  abstract.
manual: Open the CLEO 2025 planner at
  https://cleo2025.abstractcentral.com/planner.jsp , click the "Planner" link to
  reach the program, wait for it to load, and click the
  "Program + Abstracts (PDF)" download button (NOT "Program Only"). Save it into
  this folder's data/ subdirectory as CLEO2025_Program_Abstracts.pdf .

[file: CLEO2025_Program_Abstracts.csv]
required: yes
produced_by: fetch
description: Official "Program + Abstracts (Excel)" export.
manual: On the same planner program page, click "Program + Abstracts (Excel)".
  Save it into data/ as CLEO2025_Program_Abstracts.csv .

[file: CLEO2025_planner_expanded.html]
required: yes
produced_by: fetch
description: Full DOM snapshot (outerHTML) of the planner AFTER every day,
  session, and "See More…" control has been expanded.
manual: Preferred: let the fetch script create it. If doing it manually, open
  https://cleo2025.abstractcentral.com/planner.jsp , click "Planner", expand
  EVERY day and session "+" and every "See More…" until nothing is collapsed,
  then save the live DOM (browser console: copy(document.documentElement
  .outerHTML)) into data/ as CLEO2025_planner_expanded.html .

[file: CLEO2025_short_courses.html]
required: yes
produced_by: fetch
description: Raw HTML of the CLEO 2025 short-courses page.
manual: Use the archived 2025 snapshot:
  https://web.archive.org/web/20250129233143/https://cleoconference.org/shortcourses/
  Open it, let it finish loading, and save the page (Save As -> "Web Page, HTML
  only", or console: copy(document.documentElement.outerHTML)) into data/ as
  CLEO2025_short_courses.html .
